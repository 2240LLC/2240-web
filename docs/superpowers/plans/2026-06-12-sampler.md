# Persistent Sampler Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build `js/sampler.js` — a draggable, resizable floating sampler window that injects itself into every page of 2240.us, letting visitors sample audio from any release in MPC, Loop, and Slice modes.

**Architecture:** A single self-contained script (not a module) that creates the `window.AUDIO_CTX` singleton and injects both the sampler window (`#sampler-window`) and the `[S]` trigger button into the DOM on `DOMContentLoaded`. All three modes render into the same content area and share a single audio buffer per source. State is persisted to `sessionStorage` so the window position, size, and active mode survive page navigation. No framework, no build tools.

**Tech Stack:** Web Audio API (`AudioContext`, `AudioBufferSourceNode`, `GainNode`, `OfflineAudioContext` for decoding), Canvas 2D (mini-waveform, encoder knobs), `sessionStorage`, plain HTML/CSS/JS.

  { id: 'washboard', label: 'Washboard',         artist: 'Toby Brown', file: 'audio/toby-brown-washboard.mp3',           mode: 'washboard' },
  { id: 'patient',   label: 'The Patient',       artist: 'Toby Brown', file: 'audio/toby-brown-the-patient.mp3',         mode: 'patient' },
  { id: 'statues',   label: 'Is It Even Easier?',artist: 'Toby Brown', file: 'audio/toby-brown-is-it-even-easier.mp3',   mode: 'statues' },
];

const MIN_W = 320, MIN_H = 240, DEF_W = 480, DEF_H = 320;

// ── AudioContext ──────────────────────────────────────────────────────────────
function getCtx() {
  if (!window.AUDIO_CTX) window.AUDIO_CTX = new AudioContext();
  return window.AUDIO_CTX;
}

// ── Buffer cache ──────────────────────────────────────────────────────────────
const bufferCache = new Map();  // Map<id, AudioBuffer>
let activeBuffer   = null;
let activeSourceId = SOURCES[0].id;

async function loadSource(id) {
  if (bufferCache.has(id)) { activeBuffer = bufferCache.get(id); return activeBuffer; }
  const src = SOURCES.find(s => s.id === id);
  if (!src) return null;
  try {
    const res = await fetch(src.file);
    if (!res.ok) throw new Error('HTTP ' + res.status);
    const raw    = await res.arrayBuffer();
    const offCtx = new OfflineAudioContext(1, 1, 44100);
    const buffer = await offCtx.decodeAudioData(raw);
    bufferCache.set(id, buffer);
    activeBuffer = buffer;
    return buffer;
  } catch (err) {
    console.warn('[sampler] Could not load', id, err.message);
    return null;
  }
}

// ── Slice trigger (used by all three modes) ───────────────────────────────────
function triggerSlice(buffer, startSec, durationSec, opts) {
  opts = opts || {};
  const ctx  = getCtx();
  const src  = ctx.createBufferSource();
  src.buffer = buffer;
  src.playbackRate.value = opts.pitch  != null ? opts.pitch  : 1.0;
  src.loop               = opts.loop   != null ? opts.loop   : false;
  if (src.loop) { src.loopStart = startSec; src.loopEnd = startSec + durationSec; }
  const gain = ctx.createGain();
  gain.gain.value = opts.volume != null ? opts.volume : 1.0;
  src.connect(gain).connect(ctx.destination);
  src.start(0, startSec, src.loop ? undefined : durationSec);
  return { src, gain };
}

// ── Mini-waveform canvas draw ─────────────────────────────────────────────────
// opts: { color, regionStart, regionEnd }  (fractions 0–1)
function drawMiniWaveform(canvas, buffer, opts) {
  opts = opts || {};
  const dpr = window.devicePixelRatio || 1;
  canvas.width  = Math.round(canvas.offsetWidth  * dpr);
  canvas.height = Math.round(canvas.offsetHeight * dpr);
  const c2 = canvas.getContext('2d');
  const W  = canvas.width, H = canvas.height;
  c2.clearRect(0, 0, W, H);
  if (!buffer) return;

  const ch      = buffer.getChannelData(0);
  const barW    = 2 * dpr;
  const numBars = Math.max(1, Math.floor(W / barW));
  const step    = Math.max(1, Math.floor(ch.length / numBars));

  for (let i = 0; i < numBars; i++) {
    let peak = 0;
    const off = i * step;
    for (let j = 0; j < step; j++) { const v = Math.abs(ch[off + j] || 0); if (v > peak) peak = v; }
    const bh = peak * H * 0.85;
    c2.fillStyle = opts.color || 'rgba(0,210,0,0.55)';
    c2.fillRect(i * barW, (H - bh) / 2, dpr, bh);
  }

  // Region overlay + handle lines
  if (opts.regionStart != null && opts.regionEnd != null) {
    const x1 = opts.regionStart * W, x2 = opts.regionEnd * W;
    c2.fillStyle = 'rgba(0,200,0,0.13)';
    c2.fillRect(x1, 0, x2 - x1, H);
    c2.fillStyle = 'rgba(0,230,0,0.8)';
    c2.fillRect(x1 - dpr, 0, 2 * dpr, H);
    c2.fillRect(x2 - dpr, 0, 2 * dpr, H);
  }
}

// ── Time formatter ────────────────────────────────────────────────────────────
function fmtTime(s) {
  const m = Math.floor(s / 60);
  return m + ':' + Math.floor(s % 60).toString().padStart(2, '0');
}

window.SAMPLER = { triggerSlice, loadSource, SOURCES };

// ── State persistence ─────────────────────────────────────────────────────────
function saveState() {
  try { sessionStorage.setItem('sam', JSON.stringify(samplerState)); } catch (_) {}
}

function loadState() {
  try {
    var saved = JSON.parse(sessionStorage.getItem('sam') || 'null');
    if (saved) Object.assign(samplerState, saved);
  } catch (_) {}
  // Clamp to viewport
  samplerState.w = Math.max(MIN_W, Math.min(samplerState.w, window.innerWidth  * 0.9));
  samplerState.h = Math.max(MIN_H, Math.min(samplerState.h, window.innerHeight * 0.9));
  samplerState.x = Math.max(0, Math.min(samplerState.x, window.innerWidth  - samplerState.w));
  samplerState.y = Math.max(0, Math.min(samplerState.y, window.innerHeight - 48));
}

function defaultPos() {
  return {
    x: Math.max(0, window.innerWidth  - DEF_W - 28),
    y: Math.max(0, window.innerHeight - DEF_H - 68),
  };
}

function applyGeometry() {
  winEl.style.left   = samplerState.x + 'px';
  winEl.style.top    = samplerState.y + 'px';
  winEl.style.width  = samplerState.w + 'px';
  winEl.style.height = samplerState.collapsed ? '48px' : samplerState.h + 'px';
}

// ── Show / hide / toggle ──────────────────────────────────────────────────────
function showSampler() {
  if (!samplerState.x && !samplerState.y) Object.assign(samplerState, defaultPos());
  samplerState.visible = true;
  winEl.classList.add('sam-visible');
  applyGeometry();
  document.getElementById('sampler-trigger').setAttribute('aria-expanded', 'true');
  saveState();
}

function hideSampler() {
  samplerState.visible = false;
  winEl.classList.remove('sam-visible');
  document.getElementById('sampler-trigger').setAttribute('aria-expanded', 'false');
  saveState();
}

function toggleSampler() { if (samplerState.visible) hideSampler(); else showSampler(); }

// ── Collapse ──────────────────────────────────────────────────────────────────
function setCollapsed(v) {
  samplerState.collapsed = v;
  winEl.classList.toggle('sam-collapsed', v);
  winEl.style.height = v ? '48px' : samplerState.h + 'px';
  document.getElementById('sam-minimize').textContent = v ? '□' : '—';
  saveState();
}

// ── Drag ──────────────────────────────────────────────────────────────────────
function initDrag() {
  var dragging = false, ox = 0, oy = 0;
  titleBarEl.addEventListener('mousedown', function(e) {
    if (e.target.classList.contains('sam-tbar-btn')) return;
    dragging = true; ox = e.clientX - samplerState.x; oy = e.clientY - samplerState.y;
    e.preventDefault();
  });
  document.addEventListener('mousemove', function(e) {
    if (!dragging) return;
    samplerState.x = Math.max(0, Math.min(e.clientX - ox, window.innerWidth  - samplerState.w));
    samplerState.y = Math.max(0, Math.min(e.clientY - oy, window.innerHeight - 48));
    winEl.style.left = samplerState.x + 'px';
    winEl.style.top  = samplerState.y + 'px';
  });
  document.addEventListener('mouseup', function() { if (dragging) { dragging = false; saveState(); } });
}

// ── Resize ────────────────────────────────────────────────────────────────────
function initResize() {
  winEl.querySelectorAll('.sam-resize').forEach(function(handle) {
    handle.addEventListener('mousedown', function(e) {
      e.preventDefault(); e.stopPropagation();
      var dir = handle.dataset.dir;
      var sx = e.clientX, sy = e.clientY;
      var sw = samplerState.w, sh = samplerState.h;
      var spx = samplerState.x, spy = samplerState.y;
      var maxW = window.innerWidth * 0.9, maxH = window.innerHeight * 0.9;

      function onMove(e) {
        var dx = e.clientX - sx, dy = e.clientY - sy;
        if (dir.indexOf('e') >= 0) samplerState.w = Math.max(MIN_W, Math.min(sw + dx, maxW));
        if (dir.indexOf('s') >= 0) samplerState.h = Math.max(MIN_H, Math.min(sh + dy, maxH));
        if (dir.indexOf('w') >= 0) {
          var nw = Math.max(MIN_W, Math.min(sw - dx, maxW));
          samplerState.x = spx + (sw - nw); samplerState.w = nw;
        }
        if (dir.indexOf('n') >= 0) {
          var nh = Math.max(MIN_H, Math.min(sh - dy, maxH));
          samplerState.y = spy + (sh - nh); samplerState.h = nh;
        }
        applyGeometry();
      }
      function onUp() {
        saveState();
        document.removeEventListener('mousemove', onMove);
        document.removeEventListener('mouseup', onUp);
      }
      document.addEventListener('mousemove', onMove);
      document.addEventListener('mouseup', onUp);
    });
  });
}

function initDefaultSlices(buffer) {
  var d = buffer.duration / 16;
  for (var i = 0; i < 16; i++) {
    pads[i] = { startSec: i * d, durationSec: d };

  if (!activeBuffer || !pads[idx] || !pads[idx].durationSec) return;

    setTimeout(function() { padEl.classList.remove('triggered'); }, 130);

function renderMPC() {
  contentEl.innerHTML = '';
  if (!activeBuffer) { contentEl.innerHTML = '<div style="color:rgba(0,210,0,0.3);font-size:8px;padding:4px;">Loading…</div>'; return; }
  initDefaultSlices(activeBuffer);

  var wfWrap = document.createElement('div');
  wfWrap.className = 'sam-wf-wrap';
  wfWrap.style.height = '48px';
  wfWrap.style.marginBottom = '6px';

  wfWrap.appendChild(wfCanvas);
  contentEl.appendChild(wfWrap);

  var assigningPad = -1;  // pad index being assigned to (-1 = none)

  function redrawAssign() {
    drawMiniWaveform(wfCanvas, activeBuffer, { regionStart: mpcRegionStart, regionEnd: mpcRegionEnd });
  }
  redrawAssign();

  // Drag to set region on waveform
  var draggingWf = false, wfDragAnchor = 0;
  wfWrap.addEventListener('mousedown', function(e) {
    var rect = wfWrap.getBoundingClientRect();
    wfDragAnchor = (e.clientX - rect.left) / rect.width;
    mpcRegionStart = wfDragAnchor;
    mpcRegionEnd   = wfDragAnchor;
    draggingWf = true;
    e.preventDefault();
  });
  document.addEventListener('mousemove', function(e) {
    if (!draggingWf) return;
    var rect = wfWrap.getBoundingClientRect();
    var f = Math.max(0, Math.min((e.clientX - rect.left) / rect.width, 1));
    mpcRegionStart = Math.min(wfDragAnchor, f);
    mpcRegionEnd   = Math.max(wfDragAnchor, f);
    if (mpcRegionEnd - mpcRegionStart < 0.005) mpcRegionEnd = mpcRegionStart + 0.005;
    redrawAssign();
  });
  document.addEventListener('mouseup', function() { draggingWf = false; });

  // Display order: top row = pads 0-3 (key row 1234), bottom row = pads 12-15 (key row zxcv)
  var grid = document.createElement('div');
  grid.className = 'sam-pad-grid';

  for (var i = 0; i < 16; i++) {

      var padEl = document.createElement('div');
      padEl.className = 'sam-pad has-slice';
      padEl.dataset.padIdx = idx;
      padEl.setAttribute('aria-label', 'Pad ' + (idx + 1));
      padEl.innerHTML =
        '<span class="sam-pad-num">' + (idx + 1) + '</span>' +
        '<span class="sam-pad-key">' + PAD_KEYS[idx] + '</span>';

      padEl.addEventListener('mousedown', function(e) {
        e.preventDefault();
        if (assigningPad === idx) {
          // Second click: confirm assignment (already assigned on first click)
          assigningPad = -1;
          contentEl.querySelectorAll('.sam-pad').forEach(function(p) { p.classList.remove('assigning'); });
        } else if (e.shiftKey) {
          // Shift+click: assign current region to this pad
          pads[idx] = {
            startSec:   mpcRegionStart * activeBuffer.duration,
            durationSec:(mpcRegionEnd - mpcRegionStart) * activeBuffer.duration,
          };
          padEl.classList.add('assigning');
          setTimeout(function() { padEl.classList.remove('assigning'); }, 400);
        } else {
          // Normal click: trigger
          firePad(idx);
        }
      });

      grid.appendChild(padEl);
    }(i));
  }

  contentEl.appendChild(grid);
}

// ── Knob ──────────────────────────────────────────────────────────────────────
// Returns { setValue(v) }. onChange fires on drag.
function makeKnob(container, label, min, max, initVal, onChange) {
  var wrap  = document.createElement('div'); wrap.className = 'knob-wrap';
  var cvs   = document.createElement('canvas'); cvs.className = 'knob-canvas';
  cvs.width = 40; cvs.height = 40;
  var lbl   = document.createElement('div'); lbl.className = 'knob-label'; lbl.textContent = label;
  wrap.appendChild(cvs); wrap.appendChild(lbl);
  container.appendChild(wrap);

  var val = initVal;

  function draw() {
    var c2 = cvs.getContext('2d'), cx = 20, cy = 20, r = 14;
    c2.clearRect(0, 0, 40, 40);
    var a0 = 0.75 * Math.PI, a1 = 2.25 * Math.PI;
    c2.beginPath(); c2.arc(cx, cy, r, a0, a1);
    c2.strokeStyle = '#2a2a2a'; c2.lineWidth = 3; c2.stroke();
    var norm = (val - min) / (max - min);
    var af   = a0 + norm * 1.5 * Math.PI;
    c2.beginPath(); c2.arc(cx, cy, r, a0, af);
    c2.strokeStyle = 'rgba(0,210,0,0.7)'; c2.lineWidth = 3; c2.stroke();
    c2.beginPath();
    c2.arc(cx + r * Math.cos(af), cy + r * Math.sin(af), 2.5, 0, 2 * Math.PI);
    c2.fillStyle = 'rgba(0,210,0,0.9)'; c2.fill();
  }
  draw();

  var y0 = null, v0 = null;
  cvs.addEventListener('mousedown', function(e) {
    y0 = e.clientY; v0 = val;
    function onMove(e) {
      val = Math.max(min, Math.min(max, v0 + (y0 - e.clientY) * (max - min) / 100));
      draw(); onChange(val);
    }
    function onUp() {
      document.removeEventListener('mousemove', onMove);
      document.removeEventListener('mouseup', onUp);
    }
    document.addEventListener('mousemove', onMove);
    document.addEventListener('mouseup', onUp);
    e.preventDefault();
  });

  return { setValue: function(v) { val = Math.max(min, Math.min(max, v)); draw(); } };
}
